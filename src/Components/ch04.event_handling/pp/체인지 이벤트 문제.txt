파일 이름 : combo_change.js

콤보 박스를 사용하여 change 이벤트를 사용하여 다음을 처리해 주세요.
콤보 박스 선택 가능 항목 : 빵, 음료수 
빵 선택시
	목록을 ul 태그로 보여 주기 
음료수
	목록을 ol 태그로 보여 주기 

ol이나 ul 항목을 클릭하면 
클릭된 <li>의 텍스트를 alert로 표시합니다.
또는 
해당 품목에 대한 이미지를 보여 줍니다.


function App() {
  function handleItemClick(e) {
    alert(e.target.innerText); // 클릭된 <li>의 텍스트를 alert로 표시
  }

  return (
    <div>
      <h2>클릭 이벤트 테스트</h2>
      <ul style={{ cursor: "pointer", backgroundColor: "#f0f0f0" }}>
        <li onClick={handleItemClick}>아이템 1</li>
        <li onClick={handleItemClick}>아이템 2</li>
        <li onClick={handleItemClick}>아이템 3</li>
      </ul>
    </div>
  );
}
